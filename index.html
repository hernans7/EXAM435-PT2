import numpy as np import pandas as pd import matplotlib.pyplot as plt from statsmodels.stats.proportion import proportions_ztest np.random.seed(42) # ------------------------- # 1. Setup experiment # ------------------------- n_students = 1000 n_control = n_students // 2 n_treatment = n_students // 2 # True probabilities (based on your scenario) p_control = 0.28 # Paragraph format p_treatment = 0.36 # Bullet point format # ------------------------- # 2. Simulate enrollments # ------------------------- control_enrollments = np.random.binomial(1, p_control, n_control) treatment_enrollments = np.random.binomial(1, p_treatment, n_treatment) # ------------------------- # 3. Compute metrics # ------------------------- control_rate = control_enrollments.mean() treatment_rate = treatment_enrollments.mean() lift = treatment_rate - control_rate print("Control Group Enrollment Rate:", round(control_rate * 100, 2), "%") print("Treatment Group Enrollment Rate:", round(treatment_rate * 100, 2), "%") print("Lift:", round(lift * 100, 2), "percentage points") # ------------------------- # 4. Statistical significance test # ------------------------- count = np.array([control_enrollments.sum(), treatment_enrollments.sum()]) nobs = np.array([n_control, n_treatment]) stat, p_value = proportions_ztest(count, nobs) print("p-value:", p_value) if p_value < 0.05: print("Result: Statistically significant difference") else: print("Result: Not statistically significant") # ------------------------- # 5. Visualization # ------------------------- labels = ['Control (Paragraph)', 'Treatment (Bullet Points)'] rates = [control_rate, treatment_rate] plt.bar(labels, rates) plt.title("A/B Test: Course Description Format vs Enrollment Rate") plt.ylabel("Enrollment Rate") plt.ylim(0, 1) plt.show()
